int logker(double A[], int N, double felt){

bool l = false;

int u = 0;

int v = N - 1;

int i = 0;

while(u <= v && !l){

i = (u + v) / 2;

if(A[i] == felt){

l = true;

}

else if(A[i] < felt){

u = i + 1;

}

else {

v = i - 1;

}

}

if(l){

return i;

}

else {

return -1;

}
