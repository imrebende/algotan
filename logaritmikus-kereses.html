int logker(double a[], int n, double felt) {

bool l = false;

int u = 0, v = n - 1, i = 0;

while(u <= v && !l) {

i = (u + v) / 2;

if(a[i] == felt) {

l = true;

} else if(a[i] < felt) {

u = i + 1;

} else {

v = i - 1;

}

}

if(l) {

return i;

} else {

return -1;

}

}
